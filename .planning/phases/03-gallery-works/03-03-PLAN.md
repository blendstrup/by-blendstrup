    ```typescript
    import { notFound } from "next/navigation"
    import { createReader } from "@keystatic/core/reader"
    import keystaticConfig from "../../../../../keystatic.config"
    import { getTranslations } from "next-intl/server"
    import WorkDetail from "@/components/WorkDetail"

    interface WorkDetailPageProps {
      params: Promise<{ locale: string; slug: string }>
    }

    export default async function WorkDetailPage({ params }: WorkDetailPageProps) {
      const { locale, slug } = await params

      const reader = createReader(process.cwd(), keystaticConfig)
      const work = await reader.collections.works.read(slug)

      // Security: unknown or draft slug → 404 (T-3-01)
      if (!work || !work.published) {
        notFound()
      }

      const title = locale === "da" ? work.titleDa : work.titleEn
      const description = locale === "da" ? work.descriptionDa : work.descriptionEn

      const t = await getTranslations("gallery")

      return (
        <WorkDetail
          title={title}
          description={description}
          saleStatus={work.saleStatus as "available" | "sold" | "notListed"}
          images={work.images as Array<{ image: string; alt: string }>}
          ctaLabels={{
            contactToBuy: t("contactToBuy"),
            soldMessage: t("soldMessage"),
            soldCta: t("soldCta"),
          }}
        />
      )
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| URL slug → Reader.read(slug) | Slug from URL params is passed directly to the Reader API — Reader returns null for unknown slugs; the page must call notFound() on null (T-3-01) |
| work.published === false → notFound() | Draft works must never be accessible via detail page URL; even if slug is valid, an unpublished work must 404 |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-01 | Information Disclosure | /gallery/[slug] page — unknown slug | mitigate | `if (!work \|\| !work.published) notFound()` — required in acceptance criteria; grep-verifiable: `grep "notFound" src/app/\\[locale\\]/gallery/\\[slug\\]/page.tsx` |
| T-3-08 | Information Disclosure | /gallery/[slug] page — draft work accessible by slug | mitigate | `!work.published` check in the notFound guard covers this; draft slugs are also excluded from generateStaticParams so they are never pre-rendered |
| T-3-09 | Denial of Service | generateStaticParams iterating all works | accept | Works collection is small (< 100 items for foreseeable future); full static pre-render adds seconds to build, not minutes; no runtime impact |
| T-3-10 | Tampering | Link href="/custom-orders" in WorkDetail | accept | Hardcoded internal path; no user-controlled input in href construction; next-intl Link prepends locale safely |
</threat_model>
