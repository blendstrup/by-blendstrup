CREATE `src/components/NavLinks.tsx`:

```typescript
"use client"

import { Link, usePathname } from "@/i18n/navigation"
import { useTranslations } from "next-intl"

export function NavLinks() {
  const pathname = usePathname()
  const t = useTranslations("navigation")

  const links = [
    { href: "/gallery" as const, key: "gallery" },
    { href: "/shop" as const, key: "shop" },
    { href: "/contact" as const, key: "contact" },
  ]

  return (
    <nav aria-label="Main navigation" className="flex gap-8">
      {links.map(({ href, key }) => {
        const isActive = pathname === href || pathname.startsWith(`${href}/`)
        return (
          <Link
            key={href}
            href={href}
            className={
              isActive
                ? "font-medium text-sm text-ink underline decoration-terracotta transition-colors duration-150"
                : "font-medium text-sm text-stone transition-colors duration-150 hover:text-ink hover:underline hover:decoration-terracotta"
            }
          >
            {t(key)}
          </Link>
        )
      })}
    </nav>
  )
}
```

Key implementation notes:
- `"use client"` directive is required — `usePathname` and `useTranslations` are React hooks
- `usePathname` from `@/i18n/navigation` returns the locale-stripped path (e.g., `/gallery` not `/da/gallery`) — confirmed in RESEARCH.md Pattern 3
- Active check uses `pathname === href || pathname.startsWith(href + "/")` to catch both `/gallery` and `/gallery/bowl-test`
- The `href` values must be typed as `const` or as the appropriate `href` type expected by the `Link` from `@/i18n/navigation`
- SiteHeader will use this component in Plan 06 — do NOT modify SiteHeader in this plan
