```typescript
import { createReader } from "@keystatic/core/reader"
import { getTranslations } from "next-intl/server"
import keystaticConfig from "../../../../keystatic.config"
import { ShopCard } from "@/components/ShopCard"

interface ShopPageProps {
  params: Promise<{ locale: string }>
}

export default async function ShopPage({ params }: ShopPageProps) {
  const { locale } = await params
  const t = await getTranslations("shop")

  const reader = createReader(process.cwd(), keystaticConfig)
  const allWorks = await reader.collections.works.all()

  // Filter: must be published AND saleStatus === 'available'
  // This is the same filter logic tested in shop-filter.test.ts (Plan 01 Task 2)
  const forSaleWorks = allWorks.filter(
    (w) => w.entry.published === true && w.entry.saleStatus === "available",
  )

  return (
    <main className="py-24 pb-16">
      <div className="mx-auto max-w-screen-xl px-6 sm:px-12 lg:px-16">
        <h1 className="mb-12 font-serif text-5xl font-normal tracking-tight text-ink">
          {t("heading")}
        </h1>

        {forSaleWorks.length === 0 ? (
          <div className="py-24 text-center">
            <p className="font-serif text-[28px] font-normal tracking-tight text-ink">
              {t("empty.heading")}
            </p>
            <p className="mt-4 font-sans text-base font-normal text-stone">
              {t("empty.body")}
            </p>
          </div>
        ) : (
          <div className="grid grid-cols-1 gap-6 sm:grid-cols-2 lg:grid-cols-3 lg:gap-8">
            {forSaleWorks.map(({ slug, entry }) => (
              <ShopCard
                key={slug}
                slug={slug}
                entry={{
                  titleDa: entry.titleDa,
                  titleEn: entry.titleEn,
                  price: entry.price ?? "",
                  leadTime: entry.leadTime ?? "",
                  saleStatus: entry.saleStatus as "available" | "sold" | "notListed",
                  images: entry.images,
                }}
                locale={locale}
                labels={{
                  sold: t("saleStatus.sold"),
                  forSale: t("saleStatus.available"),
                  contactToBuy: t("card.contactToBuy"),
                }}
              />
            ))}
          </div>
        )}
      </div>
    </main>
  )
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Keystatic settings.yaml → contact page | contactEmail and instagramHandle values flow from git-committed YAML into rendered JSX |
| External link (Instagram) → instagram.com | href constructed from instagramHandle CMS field |
| Internal stub links → Phase 5 routes | hardcoded /contact/purchase and /custom-orders paths via next-intl Link |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-05-01 | Tampering (open redirect) | Instagram link `href="https://instagram.com/${handle}"` | mitigate | `rel="noopener noreferrer"` on `target="_blank"` link prevents tab-napping. `instagramHandle` value comes from git-committed YAML (not user-supplied at runtime), limiting injection risk. The handle is max 60 chars and cannot contain `javascript:` prefix in practice. |
| T-04-05-02 | Information Disclosure | contactEmail displayed publicly | accept | Email is intentionally public on the contact page — owner configured it for this purpose. No mitigation needed; obfuscation would break the mailto link. |
| T-04-05-03 | Tampering (XSS) | instagramHandle in href attribute | mitigate | Value is a JSX attribute string, React escapes it. Value comes from CMS YAML field with max 60 chars and plain-text validation. `javascript:` injection not possible via a short handle-shaped string. |
| T-04-05-04 | Denial of Service | Shop page rendering all available works | accept | Works collection is bounded (< 200 pieces for foreseeable future). No pagination needed. Static rendering prevents per-request overhead. |
</threat_model>
